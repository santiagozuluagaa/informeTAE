## Funcion para obtener la decodificcion de los datos

def obtain_encoding(file):
    rawdata = open(file, 'rb').read()
    result = chardet.detect(rawdata)
    charenc = result['encoding']
    return(str(charenc))

enc1 = obtain_encoding("Accidentalidad_Medellin3.csv")

## Leer datos desde archivo csv
datos = pd.read_csv("Accidentalidad_Medellin3.csv", encoding = enc1)

## Obtener el numero del dia a partir de la fecha
datos['numero_dia'] = datos.apply(lambda x: datetime.datetime.strptime(x['fecha'],'%Y-%m-%d').timetuple().tm_yday, axis=1)

## Se eliminan las columnas que no seran tenidas en cuenta para el entrenamiento del modelo
datos.drop(columns=['X','Y','periodo','dia','comuna','mes','semana','diseno','gravedad','festivo'],axis=1, inplace=True)

## Se obtiene, a partir de la fecha, el mes, ao, y dia del mes
datos[['ao','mes','dia']] = datos.fecha.str.split("-",expand=True,)

## A partir de las columnas obtenidas antes, se obtiene los dias festivos
datos['festivo'] = datos.apply(lambda x: int(date(int(x['ao']),int(x['mes']),int(x['dia'])) in co_holidays), axis=1)

## Se obtienen dos nuevas columnas con la informacion de que si el dia antes y posterios es festivo
datos['fest_desp'] = datos.apply(lambda x: int(date(int(x['ao']),int(x['mes']),int(x['dia'])) + timedelta(days=1) in co_holidays), axis=1)
datos['fest_antes'] = datos.apply(lambda x: int(date(int(x['ao']),int(x['mes']),int(x['dia'])) - timedelta(days=1) in co_holidays), axis=1)

## Se convierte la columna mes a una de tipo entero y no string
datos['mes'] = datos['mes'].astype(int)

## Se eliminan las columnas que no seran tenidas en cuenta para el entrenamiento del modelo
datos.drop(columns=['dia','fecha'], inplace = True)

## Ciclo para cambiar la decodificacion de las columnas

## Se eliminan las clases 'volcamiento' e 'incendio', por la poca cantidad de datos

mse_train_values = pd.DataFrame(columns = ['clase','mse_train_random'])

for j in range(0,4):
    
    X_class = train_data[train_data['clase_code'] == j].reset_index(drop=True)
    y_pred = X_class[['group_count']]
    X_pred = X_class[['numero_dia','festivo','fest_antes','fest_desp','dia_code','mes','clase_code','barrio_code']]
    X_pred['predictions'] = pd.Series(poisson_results.predict(X_pred))
    X_pred['predictions'] = X_pred['predictions'].round().astype(int)
    X_pred.drop(columns=['festivo','fest_antes','fest_desp','dia_code','mes','clase_code'], inplace=True)
    mse_train_random = mean_squared_error(X_pred['predictions'],y_pred)
    mse_train_values.loc[j] = [clase_enc.inverse_transform([j])[0], mse_train_random];
    
    
mse_val_values = pd.DataFrame(columns = ['clase','mse_val_random'])

for j in range(0,4):
    val_pred = val[val['clase_code'] == j].reset_index(drop=True)
    if val_pred.empty != True:
        X_val = val_pred[['numero_dia','festivo','fest_antes','fest_desp','dia_code','mes','clase_code','barrio_code']]
        y_val = val_pred[['group_count']]
        val_pred['predictions'] = pd.Series(poisson_results.predict(X_val))
        val_pred['predictions'] = val_pred['predictions'].round().astype(int)

        val_pred.drop(columns=['festivo','fest_antes','fest_desp','dia_code','mes','clase_code','barrio_code'], inplace=True)
        val_pred['fecha'] = val_pred.apply(lambda x: datetime.datetime.strptime(str(x['numero_dia']),'%j').strftime('%d/%m'), axis=1)

        mse_val_random = mean_squared_error(val_pred['group_count'],val_pred['predictions'])

        mse_val_values.loc[j] = [clase_enc.inverse_transform([j])[0], mse_val_random];
    
    else:
        
        mse_val_values.loc[j] = [clase_enc.inverse_transform([j])[0], np.nan];
