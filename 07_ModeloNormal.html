Función para obtener MSE

MSE <- function(X, Y){
  MSE <- mean((X - Y)^2)
  return(MSE)
}

Función variación porcentual

var_MSE <- function(x, y){
  var_por <- (x - y)/max(x, y)
  return(100 * abs(var_por))
}

# Función para cambiar los tipos de datos

cambio_tipo_datos <- function(df){
  # Vector con columnas a cambiar su tipo de dato
  cols <- c("mes",
            "mes_dia",
            "y_dia",
            "semana","semana_dia",
            "festivo",
            "especial",
            "fes_antes",
            "fes_despues",
            "celebracion")
  
  # Operaciones
  df[cols] <- lapply(df[cols], as.factor)
  df$fecha <- as.Date(df$fecha, "%Y-%m-%d")
  df
}

Ajuste del modelo

  • Se utilizan las variables seleccionas en con la función stepAIC, pero se incluye semana en vez de mes (más informativa)
mod_choque <- lm(formula = choque ~ semana + semana_dia + fes_antes + fes_despues + celebracion, data = train)

# Chequeo en entrenamiento
MSE_train_choque <- MSE(train$choque, fitted(mod_choque))

# Chequeo en validación
res_choque <- data.frame(fecha = test$fecha, 
                         observado = test$choque, 
                         ajustado = predict(mod_choque, newdata = test))

MSE_test_choque <- MSE(res_choque$observado, res_choque$ajustado)
